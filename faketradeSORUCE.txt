if not game:IsLoaded() then game.Loaded:Wait() end

-- ── 1. PARCHE ANTICHEAT (Synchronizer) ──────────────────────
local function WaitFor(path, timeout)
    timeout = timeout or 15
    local started = os.clock()
    local function left()
        return math.max(0, timeout - (os.clock() - started))
    end
    local segments = {}
    for seg in path:gmatch("[^%.]+") do table.insert(segments, seg) end
    local current
    local ok, svc = pcall(function() return game:GetService(segments[1]) end)
    current = (ok and svc) or game:WaitForChild(segments[1], left())
    if not current then return nil end
    for i = 2, #segments do
        current = current:WaitForChild(segments[i], left())
        if not current then return nil end
    end
    return current
end

do
    local syncModule = WaitFor("ReplicatedStorage.Packages.Synchronizer", 30)
    local Sync = {}
    if syncModule then
        pcall(function() Sync = require(syncModule) end)
    end
    if type(Sync) ~= "table" then Sync = {} end

    for _, fn in pairs(Sync) do
        if typeof(fn) ~= "function" then continue end
        if isexecutorclosure(fn) then continue end
        local ok, ups = pcall(debug.getupvalues, fn)
        if not ok then continue end
        for idx, val in pairs(ups) do
            if typeof(val) == "function" and not isexecutorclosure(val) then
                local ok2, innerUps = pcall(debug.getupvalues, val)
                if ok2 then
                    for _, v in pairs(innerUps) do
                        if typeof(v) == "boolean" then
                            debug.setupvalue(fn, idx, newcclosure(function() end))
                            break
                        end
                    end
                end
            end
        end
    end
    print("[HOP] Anticheat parcheado.")
end

-- ── 3. OBTENER SERVIDORES PUBLICOS VIA API DE ROBLOX ────────
-- Usa la funcion HTTP del executor (request / syn.request / http.request)
local function httpGet(url)
    local response

    -- Detectar que funcion HTTP tiene el executor
    if typeof(request) == "function" then
        response = request({ Url = url, Method = "GET" })
    elseif typeof(http) == "table" and typeof(http.request) == "function" then
        response = http.request({ Url = url, Method = "GET" })
    elseif typeof(syn) == "table" and typeof(syn.request) == "function" then
        response = syn.request({ Url = url, Method = "GET" })
    elseif typeof(fluxus) == "table" and typeof(fluxus.request) == "function" then
        response = fluxus.request({ Url = url, Method = "GET" })
    else
        error("No se encontro funcion HTTP en el executor")
    end

    if not response or response.StatusCode ~= 200 then
        return nil
    end
    return response.Body
end

local function getPublicServers(placeId, cursor)
    local url = string.format(
        "https://games.roblox.com/v1/games/%d/servers/Public?sortOrder=Asc&excludeFullGames=false&limit=100",
        placeId
    )
    if cursor and cursor ~= "" then
        url = url .. "&cursor=" .. cursor
    end

    local ok, body = pcall(httpGet, url)
    if not ok or not body then
        warn("[HOP] Error HTTP: " .. tostring(body))
        return nil, nil
    end

    local ok2, data = pcall(function() return HttpService:JSONDecode(body) end)
    if not ok2 or not data or not data.data then
        warn("[HOP] Error al parsear JSON")
        return nil, nil
    end

    return data.data, data.nextPageCursor
end

-- ── 4. TELEPORT A UN SERVER PUBLICO ─────────────────────────
local placeId       = game.PlaceId
local currentJobId  = game.JobId

print("[HOP] Buscando servidores publicos para PlaceId: " .. placeId)
print("[HOP] Server actual (privado): " .. currentJobId)

task.wait(1)

local function hop()
    local cursor = nil
    local intentos = 0

    repeat
        local servers, nextCursor = getPublicServers(placeId, cursor)
        cursor = nextCursor

        if not servers then
            warn("[HOP] No se pudo obtener lista, reintentando en 2s...")
            task.wait(2)
            cursor = nil
            continue
        end

        for _, server in ipairs(servers) do
            -- Saltar el server actual y servers vacios
            if server.id == currentJobId then continue end
            if (server.playing or 0) < 1 then continue end

            intentos += 1
            print(string.format("[HOP] Intentando server publico #%d (jugadores: %d, id: %s)",
                intentos, server.playing, server.id))

            local ok, err = pcall(function()
                TeleportService:TeleportToPlaceInstance(placeId, server.id, LocalPlayer)
            end)

            if ok then
                print("[HOP] Teleport enviado, esperando carga...")
                task.wait(5)
                -- Si sigue aqui, el server estaba lleno; probar el siguiente
                print("[HOP] Server lleno o fallo, probando siguiente...")
            else
                warn("[HOP] Error de teleport: " .. tostring(err))
            end
        end

        -- Si se acabaron los servers en esta pagina y no hay mas
        if not cursor or cursor == "" then
            print("[HOP] Se revisaron todos los servers disponibles, reiniciando busqueda...")
            task.wait(3)
            cursor = nil
        end

    until false
end

hop()
